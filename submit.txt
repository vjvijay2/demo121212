PROCEDURE GET_ROLE_LIST(
V_ROLE_NAME IN VARCHAR2,
L_START_PAGE IN VARCHAR2,
L_PAGE_SIZE IN VARCHAR2,
V_RES OUT VARCHAR2,
V_RESULT OUT VARCHAR2,
P_CURSOR OUT SYS_REFCURSOR
)IS
BEGIN
	OPEN P_CURSOR FOR
	SELECT 
	 ROLE_NAME,
	 ID,
	 ROLE_DESCRIPTION,
	 ROLE_STATUS,
	 CREATE_BY,
	 CREATED_DATE,
	 LAST_EDIT_BY,
	 LAST_EDIT_DATE,

	SELECT
	ROW_NUMBER()  OVER (ORDER BY CREATED_DATE DESC) AS RN,
	COUNT(1) OVER() AS  ROWCOUNT,

	 ROLE_STATUS,

	V_RESULT :='Success';

PROCEDURE ROLE_ASSIGN(
V_ROLE_ID IN VARCHAR2,
V_MENU_ID IN VARCHAR2,
V_CREATED_BY IN VARCHAR2,
V_RES OUT VARCHAR2,
V_RESULT OUT VARCHAR2
)IS 
BEGIN
	DELETE FROM DMS.C_ROLE_MENU_MANAGE_T
    WHERE ROLE_ID = V_ROLE_ID;
   
    FOR role_rec IN (
      SELECT ROLE_ID FROM DMS.C_ROLE_MENU_MANAGE_T WHERE INSTR (',' || V_ROLE_ID || ',', ',' || ROLE_ID || ',') > 0
      ) LOOP
        FOR menu_rec IN (
          SELECT MENU_ID FROM DMS.C_ROLE_MENU_MANAGE_T WHERE INSTR (',' || V_MENU_ID || ',', ',' || MENU_ID || ',') > 0
        ) LOOP 
            INSERT INTO DMS.C_ROLE_MENU_MANAGE_T (ID, ROLE_ID, MENU_ID, CREATED_BY) 
            VALUES (C_ROLE_MENU_MANAGE_T_SEQ.NEXTVAL, V_ROLE_ID, V_MENU_ID, V_CREATED_BY);
          END LOOP;
        END LOOP;
       
       COMMIT;

V_RES := '1';
V_RESULT := 'Data updated successfully ';

EXCEPTION
   WHEN OTHERS THEN
     V_RES := '0';
     V_RESULT := 'Error:' || SQLERRM;
     ROLLBACK;
END;

PROCEDURE INSERT_USER_AND_ROLE_ASSIGN(
    V_USER_ID   IN VARCHAR2,
    V_ROLE_ID   IN VARCHAR2,
    V_CREATE_BY IN VARCHAR2,
    V_RES       OUT VARCHAR2,
    V_RESULT    OUT VARCHAR2
)
IS
BEGIN
    -- Delete existing records for the given USER_ID where DEL_FLAG = 0
    DELETE FROM DMS.C_USER_AND_ROLE_MANAGE_T
    WHERE USER_ID = V_USER_ID
      AND DEL_FLAG = 0;

    -- Split and iterate over each ROLE_ID and USER_ID combination
    FOR role IN (SELECT REGEXP_SUBSTR(V_ROLE_ID, '[^,]+', 1, LEVEL) AS ROLE_ID
                 FROM DUAL
                 CONNECT BY REGEXP_SUBSTR(V_ROLE_ID, '[^,]+', 1, LEVEL) IS NOT NULL) 
    LOOP
        FOR user_rec IN (SELECT REGEXP_SUBSTR(V_USER_ID, '[^,]+', 1, LEVEL) AS USER_ID
                         FROM DUAL
                         CONNECT BY REGEXP_SUBSTR(V_USER_ID, '[^,]+', 1, LEVEL) IS NOT NULL)
        LOOP
            -- Insert new records
            INSERT INTO DMS.C_USER_AND_ROLE_MANAGE_T (ID, ROLE_ID, USER_ID, CREATE_BY)
            VALUES (C_USER_AND_ROLE_MANAGE_T_SEQ.NEXTVAL, role.ROLE_ID, user_rec.USER_ID, V_CREATE_BY);
        END LOOP;
    END LOOP;

    -- Commit the transaction
    COMMIT;

    -- Success response
    V_RES := '1';
    V_RESULT := 'Success';

EXCEPTION
    WHEN OTHERS THEN
        -- Rollback transaction in case of failure
        ROLLBACK;
        
        -- Capture the error message
        V_RES := '0';
        V_RESULT := 'Error: ' || SQLERRM;
END ;

PROCEDURE DORM_USER_ASSIGN(
    V_EMP_ID IN VARCHAR2,
    V_DATA_ARRAY IN SYS.ODCIVARCHAR2LIST,  -- Assuming array of JSON strings
    V_RES OUT VARCHAR2,
    V_RESULT OUT VARCHAR2
) AS
    v_json_obj JSON_OBJECT_T;
    v_data_control_id VARCHAR2(100);
    v_dorm_id VARCHAR2(100);
    v_check_in VARCHAR2(1);
    v_check_out VARCHAR2(1);
    v_dorm_shifting VARCHAR2(1);
    v_bed_shifting VARCHAR2(1);
    v_room_block_bed_block VARCHAR2(1);
    v_dorm_edit VARCHAR2(1);
    v_dorm_delete VARCHAR2(1);
    v_floor_edit VARCHAR2(1);
    v_floor_delete VARCHAR2(1);
    v_room_delete VARCHAR2(1);
    v_room_edit VARCHAR2(1);
    v_bed_edit VARCHAR2(1);
    v_bed_delete VARCHAR2(1);
    v_create_by VARCHAR2(100);
BEGIN
    -- First delete existing records for this employee
    DELETE FROM DMS.R_DATA_CONTROL_MENU_HANDLE_T 
    WHERE EMP_ID = V_EMP_ID;

    -- Process each JSON object in the array
    FOR i IN 1..V_DATA_ARRAY.COUNT LOOP
        v_json_obj := JSON_OBJECT_T(V_DATA_ARRAY(i));

        -- Extract values from JSON object
        v_data_control_id := v_json_obj.get_string('DATA_CONTROL_ID');
        v_dorm_id := v_json_obj.get_string('DORM_ID');
        v_check_in := v_json_obj.get_string('CHECK_IN');
        v_check_out := v_json_obj.get_string('CHECK_OUT');
        v_dorm_shifting := v_json_obj.get_string('DORM_SHIFTING');
        v_bed_shifting := v_json_obj.get_string('BED_SHIFTING');
        v_room_block_bed_block := v_json_obj.get_string('ROOM_BLOCK_BED_BLOCK');
        v_dorm_edit := v_json_obj.get_string('DORM_EDIT');
        v_dorm_delete := v_json_obj.get_string('DORM_DELETE');

 PROCEDURE GET_EMPID_BASED_MENUS (
    V_EMP_ID IN VARCHAR2,
    V_RES OUT VARCHAR2,
    V_RESULT OUT VARCHAR2,
    P_CURSOR OUT SYS_REFCURSOR
) IS
BEGIN
    OPEN P_CURSOR FOR
	-- Check if both inputs are not NULL
  SELECT
DISTINCT NUM.EMP_ID,
NUM.EMP_NAME,
NUM.DEPARTMENT,
URM.USER_ID,
MA.MENU_ID as ID, 
M.MENU_NAME,
M.PARENT_ID,
M.WEB_URL,
M.DEL_FLAG AS MENU_DEL_FLAG,
M.MENU_DESCRIPTION, 
M.MENU_STATUS,
M.MENU_TYPE
FROM
DMS.C_NEW_USER_MANAGEMENT_T NUM
LEFT JOIN DMS.C_USER_AND_ROLE_MANAGE_T URM  
ON
NUM.ID = URM.USER_ID
LEFT JOIN DMS.C_ROLE_MENU_MANAGE_T MA 
ON
URM.ROLE_ID = MA.ROLE_ID
LEFT JOIN DMS.C_MENU_ACCESS_T M
ON
M.ID = MA.MENU_ID
WHERE 
NUM.EMP_ID = V_EMP_ID
AND NUM.DEL_FLAG = 0
AND URM.DEL_FLAG = 0
AND MA.DEL_FLAG = 0
ORDER BY MA.MENU_ID;

    -- Set output variables after query execution
    V_RES := '1';
    V_RESULT := 'Success';

EXCEPTION
    WHEN OTHERS THEN
        V_RES := '0';
        V_RESULT := 'Error: ' || SQLERRM;
END ;




 PROCEDURE INSERT_USER_DATA_CONTROL (
    V_EMP_ID IN NUMBER,
    V_DORM_ID IN VARCHAR2, -- Comma-separated list of DORM_IDs
    V_CREATE_BY IN VARCHAR2,
   V_RES OUT VARCHAR2,
    V_RESULT OUT VARCHAR2
) IS
      V_DORM_CODE NUMBER;
    V_DORM_NAME VARCHAR2(100);
BEGIN
    -- Delete existing records for the given EMP_ID
    DELETE FROM DMS.C_DATA_CONTROL_T WHERE EMP_ID = V_EMP_ID;

    -- Iterate through the comma-separated DORM_ID values
    FOR DORM_ROW IN (
        SELECT REGEXP_SUBSTR(V_DORM_ID, '[^,]+', 1, LEVEL) AS DORM_ID
        FROM DUAL
        CONNECT BY REGEXP_SUBSTR(V_DORM_ID, '[^,]+', 1, LEVEL) IS NOT NULL
    ) LOOP
        -- Fetch DORM_CODE and DORM_NAME for each DORM_ID
        BEGIN
            SELECT DORM_CODE, DORM_NAME
            INTO V_DORM_CODE, V_DORM_NAME
            FROM DMS.C_DORM_CREATION_T
            WHERE ID = DORM_ROW.DORM_ID AND DEL_FLAG = 0;
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
                CONTINUE; -- Skip inserting if no dorm data found
        END;

        -- Insert new record
        INSERT INTO DMS.C_DATA_CONTROL_T (ID, EMP_ID, DORM_ID, DORM_CODE, DORM_NAME, CREATE_BY)
        VALUES (C_DATA_CONTROL_T_SEQ.NEXTVAL, V_EMP_ID, DORM_ROW.DORM_ID, V_DORM_CODE, V_DORM_NAME, V_CREATE_BY);
    END LOOP;

    -- Commit the transaction
    COMMIT;

    -- Set success response
    V_RES := '1';
    V_RESULT := 'Data inserted successfully';

EXCEPTION
    WHEN OTHERS THEN
        -- Handle errors
        ROLLBACK;
        V_RES := '0';
        V_RESULT := 'ERROR: ' || SQLERRM;
END ;

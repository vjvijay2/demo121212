PROCEDURE GET_OCCUPANCY_VACANCY_SUMMART(
 V_DORM_ID IN NUMBER,
 V_SITE IN NUMBER,
 V_EMP_ID IN VARCHAR2,
 V_INPUT_DATE IN VARCHAR2,
 L_START_PAGE IN VARCHAR2,
 L_PAGE_SIZE IN VARCHAR2,
 V_RES OUT VARCHAR2,
 V_RESULT OUT VARCHAR2,
 P_CURSOR OUT SYS_REFCURSOR
)IS 
BEGIN
    IF V_INPUT_DATE IS NOT NULL AND V_EMP_ID IS NOT NULL 
    THEN 
        OPEN P_CURSOR FOR
     SELECT 
      DORM_NAME,
            DORM_LOCATION,
            VACANT_ROOM_CAPACITY,
            OVERALL_DORM_CAPACITY,
            OVERALL_FLOOR_CAPACITY,
            OCCUPIED_FLOOR_CAPACITY,
            VACANT_FLOOR_CAPACITY,
            SUM_ROOM_CAPACITY,
            OCCUPIED_ROOM_CAPACITY,
            SUM_OF_BED_CAPACITY,
            OCCUPIED_BED_CAPACITY,
            VACANT_BED_CAPACITY,
          OCCUPIED_BED_RATIO,
           VACANT_BED_RATIO,
            RN,
            ROWCOUNT,
            CASE 
                WHEN SITE = 69941 THEN 'FXCN' 
                WHEN SITE = 95326 THEN 'FXBL' 
                ELSE TO_CHAR(SITE) -- Optional ELSE to handle other cases
            END AS SITE
     
     
     FROM (
  SELECT
     S.DORM_NAME,
       S.  DORM_LOCATION,
       DC.SITE,
          OVERALL_DORM_CAPACITY AS OVERALL_DORM_CAPACITY,
          OVERALL_FLOOR_CAPACITY AS OVERALL_FLOOR_CAPACITY,
          OCCUPIED_FLOOR_CAPACITY AS OCCUPIED_FLOOR_CAPACITY,
          VACANT_FLOOR_CAPACITY AS VACANT_FLOOR_CAPACITY,
          SUM_ROOM_CAPACITY AS SUM_ROOM_CAPACITY,
          OCCUPIED_ROOM_CAPACITY AS OCCUPIED_ROOM_CAPACITY,
          VACANT_ROOM_CAPACITY AS VACANT_ROOM_CAPACITY,
          SUM_OF_BED_CAPACITY AS SUM_OF_BED_CAPACITY,
          OCCUPIED_BED_CAPACITY AS OCCUPIED_BED_CAPACITY,
          VACANT_BED_CAPACITY AS VACANT_BED_CAPACITY,
        --  OCCUPIED_BED_RATIO AS OCCUPIED_BED_RATIO,
       --   VACANT_BED_RATIO AS VACANT_BED_RATIO,
          ROUND(OCCUPIED_BED_RATIO, 2) AS OCCUPIED_BED_RATIO,
           ROUND(VACANT_BED_RATIO, 2) AS VACANT_BED_RATIO,

            ROW_NUMBER() OVER (ORDER BY S.WORK_DATE DESC) AS RN,
                COUNT(1) OVER () AS ROWCOUNT
  FROM
    DMS.C_SUMMARY_REPORT_T S
          INNER JOIN DMS.C_DATA_CONTROL_T D ON S.DORM_ID = D.DORM_ID
    JOIN DMS.C_DORM_CREATION_T DC ON DC.ID  = S.DORM_ID 
  WHERE
    WORK_DATE = TO_DATE(V_INPUT_DATE, 'DD-MM-YYYY')
    AND S.DORM_ID = NVL(V_DORM_ID, S.DORM_ID)
    AND D.EMP_ID = V_EMP_ID
   AND DC.SITE =NVL(V_SITE,DC.SITE)
) 

       WHERE        RN > (TO_NUMBER(L_START_PAGE) - 1) * TO_NUMBER(L_PAGE_SIZE)
            AND RN <= (TO_NUMBER(L_START_PAGE) - 1) * TO_NUMBER(L_PAGE_SIZE) + TO_NUMBER(L_PAGE_SIZE);   
        V_RES := '1';
        V_RESULT := 'Success';
    ELSE
        OPEN P_CURSOR FOR
        SELECT
            DORM_NAME,
            CASE 
                WHEN SITE = 69941 THEN 'FXCN' 
                WHEN SITE = 95326 THEN 'FXBL' 
                ELSE TO_CHAR(SITE) -- Optional ELSE to handle other cases
            END AS SITE,
            DORM_LOCATION,
            VACANT_ROOM_CAPACITY,
            OVERALL_DORM_CAPACITY,
            OVERALL_FLOOR_CAPACITY,
            OCCUPIED_FLOOR_CAPACITY,
            VACANT_FLOOR_CAPACITY,
            SUM_ROOM_CAPACITY,
            OCCUPIED_ROOM_CAPACITY,
            SUM_OF_BED_CAPACITY,
            OCCUPIED_BED_CAPACITY,
            VACANT_BED_CAPACITY,
            OCCUPIED_BED_RATIO,
            VACANT_BED_RATIO,
            RN,
            ROWCOUNT
        FROM (
            SELECT
                D.DORM_NAME,
                D.SITE,
                D.DORM_LOCATION,
                COUNT(DISTINCT CASE WHEN B.CREATE_BED_FORM_STATUS NOT IN(1,3) THEN R.ID END) AS VACANT_ROOM_CAPACITY,
                D.DORM_CAPACITY AS OVERALL_DORM_CAPACITY,
                D.FLOOR_CAPACITY AS OVERALL_FLOOR_CAPACITY,
                COUNT(DISTINCT CASE WHEN B.CREATE_BED_FORM_STATUS = 1 THEN B.FLOOR_ID END) AS OCCUPIED_FLOOR_CAPACITY,
                COUNT(DISTINCT CASE WHEN F.CREATE_FLOOR_FORM_STATUS != 3 THEN F.ID END) AS VACANT_FLOOR_CAPACITY,
                (SELECT SUM(F2.ROOM_CAPACITY) FROM DMS.C_FLOOR_CREATION_T F2 WHERE F2.DORM_ID = D.ID) AS SUM_ROOM_CAPACITY,
                (SELECT SUM(F2.ROOM_CAPACITY) FROM DMS.C_FLOOR_CREATION_T F2 WHERE F2.DORM_ID = D.ID) -  
                COUNT(DISTINCT CASE WHEN B.CREATE_BED_FORM_STATUS NOT IN(1,3) THEN R.ID END) AS OCCUPIED_ROOM_CAPACITY,
                NVL((SELECT SUM(R2.BED_CAPACITY) FROM DMS.C_ROOM_CREATION_T R2 WHERE R2.DORM_ID = D.ID), 0) AS SUM_OF_BED_CAPACITY,
                COUNT(CASE WHEN B.CREATE_BED_FORM_STATUS = 1 THEN 1 END) AS OCCUPIED_BED_CAPACITY,
                COUNT(CASE WHEN B.CREATE_BED_FORM_STATUS = 0 THEN 1 END) AS VACANT_BED_CAPACITY,
                -- Fixed Occupied Bed Ratio
                CASE 
                    WHEN MAX(D.DORM_CAPACITY) > 0 
                    THEN ROUND(
                           (COUNT(DISTINCT CASE WHEN B.CREATE_BED_FORM_STATUS = 1 THEN B.ID END) / 
                           TO_NUMBER(MAX(D.DORM_CAPACITY))) * 100, 
                         2
                         )
                    ELSE 0 
                END AS OCCUPIED_BED_RATIO,
                -- Fixed Vacant Bed Ratio
                CASE 
                    WHEN MAX(D.DORM_CAPACITY) > 0 
                    THEN ROUND(
                           ((TO_NUMBER(MAX(D.DORM_CAPACITY)) - 
                            COUNT(DISTINCT CASE WHEN B.CREATE_BED_FORM_STATUS = 1 THEN B.ID END)) / 
                           TO_NUMBER(MAX(D.DORM_CAPACITY))) * 100, 
                         2
                         )
                    ELSE 0 
                END AS VACANT_BED_RATIO,
                ROW_NUMBER() OVER (ORDER BY D.CREATED_DATE DESC) AS RN,
                COUNT(1) OVER () AS ROWCOUNT
            FROM
                DMS.C_DORM_CREATION_T D
                LEFT JOIN DMS.C_FLOOR_CREATION_T F ON D.ID = F.DORM_ID AND F.DEL_FLAG = 0
                LEFT JOIN DMS.C_ROOM_CREATION_T R ON F.ID = R.FLOOR_ID AND R.DEL_FLAG = 0
                LEFT JOIN DMS.C_BED_CREATION_T B ON R.ID = B.ROOM_ID AND B.DEL_FLAG = 0
                INNER JOIN DMS.C_DATA_CONTROL_T DC ON DC.DORM_ID = D.ID
            WHERE
                DC.EMP_ID = V_EMP_ID
                AND D.ID = NVL(V_DORM_ID, D.ID)
                AND D.SITE= NVL(V_SITE,D.SITE)
                AND D.DEL_FLAG = 0
            GROUP BY
                D.DORM_NAME,
                D.DORM_LOCATION,
                D.DORM_CAPACITY,
                D.FLOOR_CAPACITY,
                D.ID,
                D.CREATED_DATE,
                D.SITE
            ORDER BY D.CREATED_DATE DESC
        )
        WHERE 
            RN > (TO_NUMBER(L_START_PAGE) - 1) * TO_NUMBER(L_PAGE_SIZE)
            AND RN <= (TO_NUMBER(L_START_PAGE) - 1) * TO_NUMBER(L_PAGE_SIZE) + TO_NUMBER(L_PAGE_SIZE);
            
        V_RES := '1';
        V_RESULT := 'Success';
    END IF;
EXCEPTION
    WHEN OTHERS THEN
        V_RES := '0';
        V_RESULT := 'Error: ' || SQLERRM;
END;
     
PROCEDURE GET_DORM_LIVE_HEAD_COUNT(
  V_EMP_ID IN VARCHAR2,
  V_SITE IN NUMBER,
  V_DORM_ID IN NUMBER,
  L_START_PAGE IN VARCHAR2,
  L_PAGE_SIZE IN VARCHAR2,
  V_RES OUT VARCHAR2,
  V_RESULT OUT VARCHAR2,
  P_CURSOR OUT SYS_REFCURSOR
) IS
BEGIN 
   OPEN P_CURSOR FOR  
   SELECT  
     RN,
     ROWCOUNT,
     DORM_NAME,
     OVERALL_DORM_CAPACITY,
     OVERALL_CHECK_IN_COUNT,
     CURRENT_CL_EMPLOYEES,
     CURRENT_HH_EMPLOYEES,
     NEW_JOINERS,
     RESIGNED_TODAY,
     ABSCOND,
     DORM_SHIFTED_TODAY,
     GAP,
        CASE 
                WHEN SITE = 69941 THEN 'FXCN' 
                WHEN SITE = 95326 THEN 'FXBL' 
                ELSE TO_CHAR(SITE) -- Optional ELSE to handle other cases
            END AS SITE,
     CASE WHEN OVERALL_DORM_CAPACITY > 0 
          THEN ROUND((OVERALL_CHECK_IN_COUNT / OVERALL_DORM_CAPACITY) * 100, 2)
          ELSE 0 
     END AS PERCENTAGE
   FROM (
     SELECT
       ROW_NUMBER() OVER (ORDER BY D.CREATED_DATE DESC) AS RN,
       D.SITE,
       COUNT(1) OVER () AS ROWCOUNT, 
       D.DORM_NAME,
       D.DORM_CAPACITY AS OVERALL_DORM_CAPACITY,
       COUNT(CASE WHEN C.CHECK_IN_FORM_STATUS IN (0, 3, 4) THEN 1 END) AS OVERALL_CHECK_IN_COUNT,
    COUNT(CASE 
         WHEN C.CHECK_IN_FORM_STATUS IN (0, 3, 4) 
              AND (C.EMPLOYEE_ID LIKE 'CL%' OR C.EMPLOYEE_ID LIKE 'SU%') 
         THEN 1 
     END) AS CURRENT_CL_EMPLOYEES,
    COUNT(CASE 
         WHEN C.CHECK_IN_FORM_STATUS IN (0, 3, 4) 
              AND NOT (C.EMPLOYEE_ID LIKE 'CL%' OR C.EMPLOYEE_ID LIKE 'SU%') 
         THEN 1 
     END) AS CURRENT_HH_EMPLOYEES,
       COUNT(CASE WHEN C.CHECK_IN_FORM_STATUS = 0 AND TRUNC(C.CHECK_IN_DATE) = TRUNC(SYSDATE - 1) THEN 1 END) AS NEW_JOINERS,
       COUNT(CASE WHEN C.CHECK_IN_FORM_STATUS = 1 AND TRUNC(C.CHECK_OUT_DATE) = TRUNC(SYSDATE - 1) AND C.EMPLOYEE_STATUS LIKE '%Resigned%' THEN 1 END) AS RESIGNED_TODAY,
       COUNT(CASE WHEN C.CHECK_IN_FORM_STATUS = 1 AND TRUNC(C.CHECK_OUT_DATE) = TRUNC(SYSDATE - 1) AND C.EMPLOYEE_STATUS LIKE '%Terminated%' THEN 1 END) AS ABSCOND,
       COUNT(CASE WHEN C.CHECK_IN_FORM_STATUS = 2 AND TRUNC(DS.LAST_EDIT_DATE) = TRUNC(SYSDATE - 1) THEN 1 END) AS DORM_SHIFTED_TODAY,
       (D.DORM_CAPACITY - COUNT(CASE WHEN C.CHECK_IN_FORM_STATUS IN (0, 3, 4) THEN 1 END)) AS GAP
     FROM
       DMS.C_DORM_CREATION_T D
     INNER JOIN
       DMS.C_DATA_CONTROL_T DC ON D.ID = DC.DORM_ID
     LEFT JOIN
       DMS.C_CHECK_IN_T C ON D.ID = C.DORM_ID
     LEFT JOIN
       DMS.C_DORM_SHIFTING_T DS ON C.ID = DS.CHECK_IN_ID AND C.CHECK_IN_FORM_STATUS = 2
     WHERE
       DC.EMP_ID = V_EMP_ID

PROCEDURE GET_DORM_SHIFT_REPORT (

    ID,
    EMPLOYEE_ID,
    EMPLOYEE_NAME,
   CASE

        WHEN SITE = 95326 THEN 'FXBL'

    CURRENT_DORM,
    CURRENT_FLOOR,

PROCEDURE GET_DORM_BULK_SHIFT_REPORT(
 V_EMP_ID IN VARCHAR2,
V_EMPLOYEE_ID IN VARCHAR2,
V_CURRENT_DORM_ID IN NUMBER,
V_START_DATE IN VARCHAR2,
V_END_DATE IN VARCHAR2,
L_START_PAGE IN VARCHAR2,
L_PAGE_SIZE IN VARCHAR2,
V_RES  OUT VARCHAR2,
V_RESULT OUT VARCHAR2,
P_CURSOR OUT SYS_REFCURSOR
) IS
BEGIN
	OPEN P_CURSOR FOR
	
	 SELECT 
            RN,
            ROWSCOUNT,
            TEMP.ID,
            EMPLOYEE_ID,
            CDI.DORM_NAME AS CURRENT_DORM_NAME,
            CDI.FLOOR_NUMBER AS CURRENT_FLOOR_NUMBER,
            CDI.ROOM_NUMBER AS CURRENT_ROOM_NUMBER,
            CDI.BED_NUMBER AS CURRENT_BED_NUMBER,
            SDI.DORM_NAME AS SHIFTING_DORM_NAME,
            SDI.FLOOR_NUMBER AS SHIFTING_FLOOR_NUMBER,
            SDI.ROOM_NUMBER AS SHIFTING_ROOM_NUMBER,
            SDI.BED_NUMBER AS SHIFTING_BED_NUMBER,
            TEMP.REASON,
            TEMP.SHIFTED_BY,
            TEMP.SHIFTED_DATE,
            TEMP.CREATED_DATE 
        FROM (
            SELECT
            ROW_NUMBER () OVER (ORDER BY B.SHIFTED_DATE DESC) AS RN,
            COUNT(B.ID) OVER () AS ROWSCOUNT,
            B.ID,
            B.EMPLOYEE_ID,
            B.CURRENT_DORM_ID,
            B.CURRENT_FLOOR_ID,
            B.CURRENT_ROOM_ID,
            B.CURRENT_BED_ID,
            B.SHIFTING_DORM_ID,
            B.SHIFTING_FLOOR_ID,
            B.SHIFTING_ROOM_ID,
            B.SHIFTING_BED_ID,
            B.REASON,
            B.SHIFTED_BY,
            TO_CHAR(B.SHIFTED_DATE,'DD-MM-YYYY HH24:MI:SS') AS SHIFTED_DATE,
            TO_CHAR(B.CREATED_DATE,'DD-MM-YYYY HH24:MI:SS') AS CREATED_DATE
           FROM
          DMS.C_DATA_CONTROL_T DC
      INNER JOIN 
            DMS.C_BULK_SHIFTING_T B ON DC.DORM_ID  = B.CURRENT_DORM_ID
        WHERE 
            1=1
            AND DC.EMP_ID  = V_EMP_ID
            AND (SHIFTED_DATE >= TO_DATE(V_START_DATE, 'DD-MM-YYYY') OR V_START_DATE IS NULL)
            AND (SHIFTED_DATE <= TO_DATE(V_END_DATE, 'DD-MM-YYYY') + 1 OR V_END_DATE IS NULL)
            AND B.EMPLOYEE_ID = NVL(V_EMPLOYEE_ID,B.EMPLOYEE_ID)
            AND B.CURRENT_DORM_ID = NVL (V_CURRENT_DORM_ID,B.CURRENT_DORM_ID)
        ) TEMP
        JOIN 
            DMS.C_BED_CREATION_T CDI ON CDI.ID = TEMP.CURRENT_BED_ID AND CDI.DEL_FLAG=0
        JOIN 
            DMS.C_BED_CREATION_T SDI ON SDI.ID = TEMP.SHIFTING_BED_ID AND SDI.DEL_FLAG=0
        WHERE 
            RN > (L_START_PAGE - 1) * L_PAGE_SIZE
            AND RN <= (L_START_PAGE - 1) * L_PAGE_SIZE + L_PAGE_SIZE;

   V_RES := '1';
	V_RESULT := 'Success';
END;



PROCEDURE GET_DASHBOARD_RESPORT(
V_EMP_ID IN VARCHAR2,
V_SITE IN NUMBER,
V_RES OUT VARCHAR2,
V_RESULT OUT VARCHAR2,
P_CURSOR OUT SYS_REFCURSOR

)IS
BEGIN
	OPEN P_CURSOR FOR
SELECT
  SITE,
  FLOOR_CAPACITY,
DORM_NAME,
DORM_ADDRESS,
DORM_ID,
DORM_CODE,
DORM_LOCATION AS LOCATION,
DORM_CAPACITY AS CAPACITY_OF_DORM,
FILE_NAME,
MIMETYPE,
IMG_SIZE,
IMG_PATH,
DORM_FLAG AS DORM_DEL_FLAG,
DATA_CONTROL_DEL_FLAG AS USER_DEL_FLAG,
BED_COUNT AS "BED_COUNT",
FLOOR_COUNT AS "FLOORCOUNT",
ROOM_COUNT AS "ROOMCOUNT",
OCCUPIED_COUNT_BED AS "OCCUPIED_COUNT_BED",
VACANT_COUNT_BED AS "VACANT_COUNT_BED",
OCCUPIED_BED_RATIO AS "OCCUPIED_BED_RATIO",
VACANT_BED_RATIO AS "VACANT_BED_RATIO"
FROM (
  SELECT
      DC.SITE,
      DC.FLOOR_CAPACITY,
      DC.DORM_NAME,
      DC.DORM_ADDRESS,
      D.DORM_ID,
      DC.DORM_CODE,
      DC.DORM_LOCATION,
      DC.DORM_CAPACITY,
      DC.FILE_NAME,
      DC.MIMETYPE,
      DC.IMG_SIZE,
      DC.IMG_PATH,
      DC.CREATE_DORM_FORM_STATUS AS DORM_FLAG,
      D.DEL_FLAG AS DATA_CONTROL_DEL_FLAG,
      DC.DORM_CAPACITY AS BED_COUNT,
      F.FLOOR_COUNT,
      R.ROOM_COUNT,
      B.OCCUPIED_COUNT_BED,
      DC.DORM_CAPACITY - B.OCCUPIED_COUNT_BED AS VACANT_COUNT_BED,
      ROUND((B.OCCUPIED_COUNT_BED / DC.DORM_CAPACITY) * 100, 2) AS OCCUPIED_BED_RATIO,
      ROUND(((DC.DORM_CAPACITY - B.OCCUPIED_COUNT_BED) / DC.DORM_CAPACITY) * 100, 2) AS VACANT_BED_RATIO
  FROM
      DMS.C_DATA_CONTROL_T D
  JOIN DMS.C_DORM_CREATION_T DC ON D.DORM_ID = DC.ID AND DC.DEL_FLAG =0
  LEFT JOIN (
      SELECT F.DORM_ID, COUNT(DISTINCT F.ID) AS FLOOR_COUNT
      FROM DMS.C_FLOOR_CREATION_T F WHERE F.DEL_FLAG=0
      GROUP BY F.DORM_ID
  ) F ON DC.ID = F.DORM_ID
  LEFT JOIN (
      SELECT R.DORM_ID, COUNT(DISTINCT R.ID) AS ROOM_COUNT
      FROM DMS.C_ROOM_CREATION_T R WHERE R.DEL_FLAG =0
      GROUP BY R.DORM_ID 
  ) R ON DC.ID = R.DORM_ID
  LEFT JOIN (
      SELECT
          B.DORM_ID,
          COUNT(DISTINCT B.ID) AS OCCUPIED_COUNT_BED
      FROM DMS.C_BED_CREATION_T B
      WHERE B.CREATE_BED_FORM_STATUS = 1 AND B.DEL_FLAG =0
      GROUP BY B.DORM_ID
  ) B ON DC.ID = B.DORM_ID
  WHERE
      D.DEL_FLAG = 0
      AND D.EMP_ID = V_EMP_ID
      AND DC.SITE = NVL(V_SITE,DC.SITE)
  ORDER BY DC.CREATED_DATE DESC	);
 
 
 V_RES:='1';
 V_RESULT :='Success';

END ;


PROCEDURE GET_DASHBOARD_POPUP (
V_DORM_ID IN  NUMBER,
V_RES OUT  VARCHAR2,
V_RESULT OUT VARCHAR2,
P_CURSOR OUT SYS_REFCURSOR

)IS
BEGIN
	OPEN P_CURSOR FOR
 SELECT 
    d.DORM_NAME,
    d.DORM_LOCATION,
    d.DORM_CAPACITY,
    d.DORM_ADDRESS,
    d.FLOOR_CAPACITY,
    COUNT(f.ID) AS CREATED_FLOOR_COUNT,
    LISTAGG(f.FLOOR_NUMBER, ', ') WITHIN GROUP (ORDER BY f.FLOOR_NUMBER) AS CREATED_FLOOR_NUMBERS,
    (SELECT SUM(f.ROOM_CAPACITY) FROM DMS.C_FLOOR_CREATION_T f WHERE f.DORM_ID = d.ID AND f.DEL_FLAG =0) AS SUM_ROOM_CAPACITY,
    f.ID AS FLOOR_ID,
    f.FLOOR_NUMBER,
    f.ROOM_CAPACITY AS FLOOR_BASED_ROOM_CAPACITY,
    COUNT(r.ID) AS FLOOR_BASED_ROOM_COUNT,
    LISTAGG(r.ROOM_NUMBER, ', ') WITHIN GROUP (ORDER BY r.ROOM_NUMBER) AS FLOOR_BASED_ROOM_NUMBERS,
    r.ID AS ROOM_ID,
    r.ROOM_NUMBER, 
    r.BED_CAPACITY AS ROOM_ID_BASED_BED_CAPACITY,
    LISTAGG(b.BED_NUMBER, ', ') WITHIN GROUP (ORDER BY b.BED_NUMBER) AS ROOM_ID_BASED_BED_NUMBERS,
  
    r.BED_CAPACITY - COUNT(CASE WHEN b.CREATE_BED_FORM_STATUS = 1 THEN b.ID END) AS AVAILABLE_BEDS,
    COUNT(CASE WHEN b.CREATE_BED_FORM_STATUS = 1 THEN b.ID END) AS BOOKED_BEDS
FROM DMS.C_DORM_CREATION_T d
LEFT JOIN DMS.C_FLOOR_CREATION_T f ON d.ID = f.DORM_ID AND f.DEL_FLAG=0
LEFT JOIN DMS.C_ROOM_CREATION_T r ON f.ID = r.FLOOR_ID AND r.DEL_FLAG=0 AND r.CREATE_ROOM_FORM_STATUS IN (0, 1, 3,4)
LEFT JOIN DMS.C_BED_CREATION_T b ON r.ID = b.ROOM_ID AND b.DEL_FLAG =0   AND b.CREATE_BED_FORM_STATUS IN (0, 1, 3,4)
WHERE d.ID = V_DORM_ID AND d.DEL_FLAG=0
    --  AND f.FLOOR_NUMBER  =V_FLOOR_NO
GROUP BY
    d.ID,
    d.DORM_NAME,
    d.DORM_LOCATION,
    d.DORM_CAPACITY,
    d.FLOOR_CAPACITY,
    f.ID,
    f.FLOOR_NUMBER,
    f.ROOM_CAPACITY, 
    r.ID,
    r.BED_CAPACITY,
    r.ROOM_NUMBER,
    d.DORM_ADDRESS 
ORDER BY d.ID, f.ID, r.ROOM_NUMBER;
V_RES :='1';
V_RESULT :='Success';
END ;

 PROCEDURE GET_LAYOUT_DORM_FLOOR_BED (
    V_DORM_ID IN NUMBER,
    V_FLOOR_NUMBER IN NUMBER,
    V_RES OUT VARCHAR2,
    V_RESULT OUT VARCHAR2,
    P_CURSOR OUT SYS_REFCURSOR
) IS
BEGIN
    -- Check if both inputs are not NULL
    IF V_DORM_ID IS NOT NULL AND V_FLOOR_NUMBER IS NOT NULL THEN
        OPEN P_CURSOR FOR 
        SELECT DISTINCT 
            BC.*, 
            D.CREATE_DORM_FORM_STATUS AS DORM_DEL_FLAG,
            FC.CREATE_FLOOR_FORM_STATUS AS FLOOR_DEL_FLAG,
            R.CREATE_ROOM_FORM_STATUS AS ROOM_DEL_FLAG
        FROM DMS.C_FLOOR_CREATION_T FC
        INNER JOIN DMS.C_BED_CREATION_T BC ON FC.ID = BC.FLOOR_ID AND BC.DEL_FLAG = 0
        INNER JOIN DMS.C_ROOM_CREATION_T R ON R.ID = BC.ROOM_ID AND R.DEL_FLAG = 0
        INNER JOIN DMS.C_DORM_CREATION_T D ON D.ID = BC.DORM_ID AND D.DEL_FLAG = 0
        WHERE FC.FLOOR_NUMBER = V_FLOOR_NUMBER
        AND FC.DORM_ID = V_DORM_ID
        AND FC.DEL_FLAG = 0
        ORDER BY BC.BED_NUMBER, BC.ROOM_NUMBER, BC.FLOOR_NUMBER;
        
    ELSE
        OPEN P_CURSOR FOR 
        SELECT DISTINCT 
            BC.FLOOR_NUMBER,
            BC.FLOOR_ID
        FROM DMS.C_BED_CREATION_T BC
        WHERE BC.DORM_ID = V_DORM_ID AND BC.DEL_FLAG =0
        ORDER BY BC.FLOOR_NUMBER ;
    END IF;

    -- Set output variables after query execution
    V_RES := '1';
    V_RESULT := 'Success';

EXCEPTION
    WHEN OTHERS THEN
        V_RES := '0';
        V_RESULT := 'Error: ' || SQLERRM;
END;

PROCEDURE GET_FILTERDORM_DORMSHIFT (
    V_DORM_ID IN NUMBER,
    V_RES OUT VARCHAR2,
    V_RESULT OUT VARCHAR2,
    P_CURSOR OUT SYS_REFCURSOR
) IS
V_SITE NUMBER;
BEGIN
    SELECT SITE INTO V_SITE FROM DMS.C_DORM_CREATION_T WHERE ID = V_DORM_ID;
    
    OPEN P_CURSOR FOR 
        SELECT
            D.*,
            D.DORM_CAPACITY AS "Bed_Count",
            booked_beds,
            CASE 
                WHEN D.DORM_CAPACITY = 0 THEN 100
                WHEN booked_beds IS NULL THEN 100
                ELSE ROUND(((D.DORM_CAPACITY - NVL(booked_beds, 0)) / D.DORM_CAPACITY) * 100, 2)
            END AS "Count_Vacancy_Ratio"
        FROM DMS.C_DORM_CREATION_T D
        LEFT JOIN (
            SELECT DORM_NAME, COUNT(*) AS COUNT_BEDS, DORM_ID
            FROM DMS.C_BED_CREATION_T B 
            WHERE B.DEL_FLAG = 0
            GROUP BY DORM_NAME, DORM_ID
        ) B ON D.ID = B.DORM_ID
        LEFT JOIN ( 
            SELECT DORM_ID, COUNT(*) AS booked_beds
            FROM DMS.C_BED_CREATION_T
            WHERE CREATE_BED_FORM_STATUS = 1 AND DEL_FLAG = 0
            GROUP BY DORM_ID
        ) DelFlag0 ON D.ID = DelFlag0.DORM_ID
        WHERE D.ID NOT IN (V_DORM_ID) AND D.DEL_FLAG = '0';
    
    V_RES := '1';
    V_RESULT := 'Success';

EXCEPTION
    WHEN OTHERS THEN
        V_RES := '0';
        V_RESULT := 'Error: ' || SQLERRM;
END;
END DORM_REPORTS;

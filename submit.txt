import React, { useState, useEffect, useCallback } from 'react';
import {
  Autocomplete,
  Button,
  CircularProgress,
  FormControl,
  Grid,
  InputLabel,
  MenuItem,
  Paper,
  Select,
  Table,
  TableBody,
  TableCell,
  TableContainer,
  TableHead,
  TablePagination,
  TableRow,
  TextField
} from '@mui/material';
import { DemoContainer } from '@mui/x-date-pickers/internals/demo';
import { LocalizationProvider } from '@mui/x-date-pickers/LocalizationProvider';
import { AdapterDayjs } from '@mui/x-date-pickers/AdapterDayjs';
import { DatePicker } from '@mui/x-date-pickers/DatePicker';
import axios from 'axios';
import * as XLSX from 'xlsx';
import dayjs from 'dayjs';
import { useDispatch, useSelector } from 'react-redux';
import { makeStyles } from '@material-ui/core/styles';
// import { fetchOccupancyvacancysummary } from '../Apiconfig/apiforreports';
// import { fetchdashboarddetailsbasedempid } from '../Apiconfig/apifordashboard';
import { decrypt, iwxKey, getdk, iwxPost, getuserAccount } from '../EncryptDecrypt/Encryptfile';
import { fetchDataforUser } from '../../Redux/Actions';
import { fetchallSiteDetails } from '../Apiconfig/apifetchfordormitory';

const columns = [
  { id: 'Site', label: 'Site' },
  { id: 'DormName', label: 'Dorm Name' },
  { id: 'DormLocation', label: 'Dorm Location' },
  { id: 'OverallDormCapacity', label: 'Dorm Capacity' },
  { id: 'OverallFloorCapacity', label: 'Overall Floor Capacity' },
  { id: 'OccupiedFloorCapacity', label: 'Occupied Floor Capacity' },
  { id: 'VacantFloorCapacity', label: 'Vacant Floor Capacity' },
  { id: 'SumRoomCapacity', label: 'Overall Room Capacity' },
  { id: 'OccupiedRoomCapacity', label: 'Occupied Room Capacity' },
  { id: 'VacantRoomCapacity', label: 'Vacant Room Capacity' },
  { id: 'SumOfBedCapacity', label: 'Overall Bed Capacity' },
  { id: 'OccupiedBedCapacity', label: 'Occupied Bed Capacity' },
  { id: 'OccupiedBedRatio', label: 'Occupied Bed Ratio' },
  { id: 'VacantBedCapacity', label: 'Vacant Bed Capacity' },
  { id: 'VacantBedRatio', label: 'Vacant Bed Ratio' },
];

const apiHeaders = {
  apiID: 'ffff-1743478123230-1020912129-2188',
  userKey: '10A33A67A312E649D907BBC927995B8A'
};

function Occupancyandvacancy() {
  const account = getuserAccount();
  const [data, setData] = useState([]);
  const [dorms, setDorms] = useState([]);
  const [dormName, setDormName] = useState(null);
  const [dormId, setDormId] = useState('');
  const [page, setPage] = useState(0);
  const [rowsPerPage, setRowsPerPage] = useState(10);
  const [selectDate, setSelectDate] = useState(null);
  const [siteList, setSiteList] = useState([]);
  const [selectedSite, setSelectedSite] = useState(null);
  const [loading, setLoading] = useState(false);
  const [exportButtonLoading, setExportButtonLoading] = useState(false);
  const [pageTotal, setPageTotal] = useState(0);

  useEffect(() => {
    const fetchSiteData = async () => {
      try {
        const res = await fetchallSiteDetails();
        setSiteList(res);
      } catch (error) {
        console.error('Error fetching site data:', error);
      }
    };
    fetchSiteData();
  }, []);

  const fetchDormData = useCallback(async () => {
    if (!selectedSite) return;

    try {
      setLoading(true);
      const params = {
        EmpId: account,
        Site: selectedSite.Id,
        Func: 'Reports-Dashboard_report_new'
      };

      const response = await iwxPost(params, { headers: apiHeaders }, true);

      if (response.IsOK === '1') {
        setDorms(response.List);
      }
    } catch (error) {
      console.error('Error fetching dorm data:', error);
    } finally {
      setLoading(false);
    }
  }, [account, selectedSite]);

  useEffect(() => {
    fetchDormData();
  }, [fetchDormData]);

  const handleSelectedSiteChange = (e, value) => {
    setDormName(null);
    setDorms([]);
    setDormId('');
    setSelectedSite(value);
  };

  const calculatedData = data?.map((item) => ({
    DORM_NAME: item.DormName,
    DORM_LOCATION: item.DormLocation,
    DORM_CAPACITY: item.OverallDormCapacity,
    OverallFloorCapacity: item.OverallFloorCapacity,
    OccupiedFloorCapacity: item.OccupiedFloorCapacity,
    VacantFloorCapacity: item.VacantFloorCapacity,
    OverallRoomCapacity: item.SumRoomCapacity,
    OccupiedRoomCapacity: item.OccupiedRoomCapacity,
    Vacant_Room_Capacity: item.VacantRoomCapacity,
    OverallBedCapacity: item.SumOfBedCapacity,
    OccupiedBedCapacity: item.OccupiedBedCapacity,
    overallbedratio: item.OccupiedBedRatio,
    vacantbedratio: item.VacantBedRatio,
    VacantBedCapacity: item.VacantBedCapacity
  }));

  const total = {
    DORM_CAPACITY: calculatedData?.reduce((acc, item) => acc + item.DORM_CAPACITY, 0),
    OverallFloorCapacity: calculatedData?.reduce((acc, item) => acc + item.OverallFloorCapacity, 0),
    OccupiedFloorCapacity: calculatedData?.reduce((acc, item) => acc + item.OccupiedFloorCapacity, 0),
    VacantFloorCapacity: calculatedData?.reduce((acc, item) => acc + item.VacantFloorCapacity, 0),
    OverallRoomCapacity: calculatedData?.reduce((acc, item) => acc + item.OverallRoomCapacity, 0),
    OccupiedRoomCapacity: calculatedData?.reduce((acc, item) => acc + item.OccupiedRoomCapacity, 0),
    VacantRoomCapacity: calculatedData?.reduce((acc, item) => acc + item.Vacant_Room_Capacity, 0),
    OverallBedCapacity: calculatedData?.reduce((acc, item) => acc + item.OverallBedCapacity, 0),
    OccupiedBedCapacity: calculatedData?.reduce((acc, item) => acc + item.OccupiedBedCapacity, 0),
    VacantBedCapacity: calculatedData?.reduce((acc, item) => acc + item.VacantBedCapacity, 0),
    overallbedratio: (
      (calculatedData.reduce((acc, item) => acc + item.OccupiedBedCapacity, 0) /
        calculatedData.reduce((acc, item) => acc + item.DORM_CAPACITY, 0)) *
      100
    ).toFixed(2),
    vacantbedratio: (
      (calculatedData.reduce((acc, item) => acc + item.VacantBedCapacity, 0) /
        calculatedData.reduce((acc, item) => acc + item.DORM_CAPACITY, 0)) *
      100
    ).toFixed(2),
  };

  const storeExportName = async (reportName) => {
    const loginLogId = sessionStorage.getItem('loginlogID');
    if (!loginLogId) return;

    const decryptLoginLog = decrypt(loginLogId, getdk(iwxKey));

    try {
      const headers = {
        // apiID:'ffff-1709630746197-1020912128-1494',
        apiID: 'ffff-1743483548846-1020912129-2318',
        userKey: '10A33A67A312E649D907BBC927995B8A'
      }
      const params = {
        EmpId: getuserAccount(),
        ReportName: reportName,
        LogIn: decryptLoginLog,
        Func: 'UserManagement-Insert_report_log_new'
      }
      const response = await iwxPost(params, { headers: headers }, true)
    } catch (error) {
      console.error('Error storing export name:', error);
    }
  }

  const handleChangePage = (event, newPage) => {
    setPage(newPage);
    handleSearch(newPage, null);
  };

  const handleChangeRowsPerPage = (event) => {
    const newRowsPerPage = +event.target.value;
    setRowsPerPage(newRowsPerPage);
    setPage(0);
    handleSearch(null, newRowsPerPage);
  };

  const handleDormChange = (e, value) => {
    setDormName(value);
    setDormId(value?.DormId || '');
  };

  const handleSelectDateChange = (date) => {
    setSelectDate(date);
  };

  const handleSearch = useCallback(async (newPage = null, pageSize = null, resetPage = false) => {
    try {
      setLoading(true);
      const pg = newPage !== null ? newPage : page;
      const pageSi = pageSize !== null ? pageSize : rowsPerPage;

      const params = {
        EmpId: account,
        Site: selectedSite?.Id || '',
        DormId: dormId || '',
        InputDate: selectDate ? dayjs(selectDate).format('DD-MM-YYYY') : '',
        StartPage: resetPage ? 1 : pg + 1,
        PageSize: pageSi,
        Func: "Reports-Occupancy_vacancy_summary_new"
      };
      console.log(params)

      const response = await iwxPost(params, { headers: apiHeaders }, true);

      if (response.IsOK === '1') {
        setData(response.List);
        console.log(response.List);
        setPageTotal(response.List.length > 0 ? response.List[0].Rowcount : 0);
        if (resetPage) setPage(0);
      }
    } catch (error) {
      console.error('Error searching data:', error);
      setData([]);
    } finally {
      setLoading(false);
    }
  }, [account, selectedSite, dormId, page, rowsPerPage, selectDate]);

  const handleExport = async () => {
    if (data?.length === 0) return;

    try {
      const params = {
        EmpId: account,
        Site: selectedSite?.Id || '',
        DormId: dormId || '',
        InputDate: selectDate ? dayjs(selectDate).format('DD-MM-YYYY') : '',
        StartPage: 1,
        PageSize: 999999999999,
        Func: "Reports-Occupancy_vacancy_summary_new"
      };
      //  console.log(params)

      const response = await iwxPost(params, { headers: apiHeaders }, true);

      const data = response.List;

      setExportButtonLoading(true);
      const exportData = data.map((row) => ({
        Site: row.Site,
        DormName: row.DormName,
        Dormlocation: row.DormLocation,
        DormCapacity: row.OverallDormCapacity,
        OverallFloorCapacity: row.OverallFloorCapacity,
        OccupiedFloorCapacity: row.OccupiedFloorCapacity,
        VacantFloorCapacity: row.VacantFloorCapacity,
        OverallRoomCapacity: row.SumRoomCapacity,
        OccupiedRoomCapacity: row.OccupiedRoomCapacity,
        VacantRoomCapacity: row.VacantRoomCapacity,
        OverallBedCapacity: row.SumOfBedCapacity,
        OccupiedBedCapacity: row.OccupiedBedCapacity,
        occupied_bed_ratio: parseFloat(row.OccupiedBedRatio).toFixed(1),
        VacantBedCapacity: row.VacantBedCapacity,
        vacantbedratio: parseFloat(row.VacantBedRatio).toFixed(1)
      }));

      exportData.push({

        DormName: 'Total',
        Dormlocation: '',
        DormCapacity: data.reduce((acc, item) => acc + item.OverallDormCapacity, 0),
        OverallFloorCapacity: data.reduce((acc, item) => acc + item.OverallFloorCapacity, 0),
        OccupiedFloorCapacity: data.reduce((acc, item) => acc + item.OccupiedFloorCapacity, 0),
        VacantFloorCapacity: data.reduce((acc, item) => acc + item.VacantFloorCapacity, 0),
        OverallRoomCapacity: data.reduce((acc, item) => acc + Number(item.SumRoomCapacity || 0), 0),
        OccupiedRoomCapacity: data.reduce((acc, item) => acc + Number(item.OccupiedRoomCapacity || 0), 0),
        VacantRoomCapacity: data.reduce((acc, item) => acc + item.VacantRoomCapacity, 0),
        OverallBedCapacity: data.reduce((acc, item) => acc + item.SumOfBedCapacity, 0),
        occupied_bed_ratio: total.overallbedratio,
        VacantBedCapacity: data.reduce((acc, item) => acc + item.VacantBedCapacity, 0),
        OccupiedBedCapacity: data.reduce((acc, item) => acc + item.OccupiedBedCapacity, 0),
        vacantbedratio: total.vacantbedratio,
      },

      );

      const workbook = XLSX.utils.book_new();
      const worksheet = XLSX.utils.json_to_sheet(exportData);
      XLSX.utils.book_append_sheet(workbook, worksheet, 'Sheet1');
      const excelBuffer = XLSX.write(workbook, { bookType: 'xlsx', type: 'array' });
      const blobData = new Blob([excelBuffer], { type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' });

      const url = URL.createObjectURL(blobData);
      const link = document.createElement('a');
      link.href = url;
      link.download = 'OccupancyvacancySummary.xlsx';
      link.click();
      URL.revokeObjectURL(url);

      await storeExportName(link.download);
    } catch (error) {
      console.error('Error exporting data:', error);
    } finally {
      setExportButtonLoading(false);
    }
  };

  return (
    <>
      <h4>Occupancy and Vacancy Summary</h4>
      <Grid container spacing={2} direction="row" sx={{ justifyContent: "flex-end", alignItems: "baseline" }}>
        <Grid item>
          <Button
            size='medium'
            variant="contained"
            onClick={() => handleSearch(null, null, true)}
            sx={{ textTransform: "none" }}
            disabled={loading}
          >
            {loading ? <CircularProgress size={24} /> : 'Search'}
          </Button>
        </Grid>
        <Grid item>
          <Button
            size='medium'
            sx={{ textTransform: "none" }}
            variant="contained"
            disabled={data.length === 0 || exportButtonLoading}
            onClick={handleExport}
          >
            {exportButtonLoading ? <CircularProgress size={24} /> : 'Export'}
          </Button>
        </Grid>
      </Grid>

      <Grid container spacing={1} sx={{ mt: 1 }}>
        <Grid item xs={12} sm={6} md={3} lg={2}>
          <Autocomplete
            options={siteList}
            autoHighlight
            ListboxProps={{ style: { maxHeight: 200 } }}
            getOptionLabel={site => site.BasicCode || ''}
            isOptionEqualToValue={(option, value) => option.Id === value.Id}
            value={selectedSite}
            onChange={handleSelectedSiteChange}
            renderInput={params => (
              <TextField {...params} fullWidth label="Site" size='small' />
            )}
            renderOption={(props, site) => (
              <MenuItem {...props}>{site.BasicCode.toString()}</MenuItem>
            )}
          />
        </Grid>
        <Grid item xs={12} sm={6} md={3} lg={2}>
          <Autocomplete
            options={dorms}
            autoHighlight
            ListboxProps={{ style: { maxHeight: 200 } }}
            getOptionLabel={dorm => dorm.DormName || ''}
            value={dormName}
            onChange={handleDormChange}

            renderInput={params => (
              <TextField
                {...params}
                label="Select the dorm"
                size='small'
                fullWidth
              />
            )}
            renderOption={(props, dorm) => (
              <MenuItem
                {...props}
                style={{
                  color: dorm.DormDelFlag === 2 || dorm.DormDelFlag === 1 ? '#FF0000' : 'initial',
                  fontWeight: dorm.DormDelFlag === 2 || dorm.DormDelFlag === 1 ? 'bold' : 'normal',
                }}
              >
                {dorm.DormName} {''}
                {dorm.DormDelFlag === 2 ? '(blocked)' : dorm.DormDelFlag === 1 ? '(deactivated)' : ''}
              </MenuItem>
            )}
          />
        </Grid>

        <Grid item xs={12} sm={6} md={3} lg={2}>
          <LocalizationProvider dateAdapter={AdapterDayjs}>
            <DatePicker
              label="Select the Date"
              value={selectDate}
              onChange={handleSelectDateChange}
              format="DD-MM-YYYY"
              slotProps={{ textField: { size: 'small', fullWidth: true } }}
            />
          </LocalizationProvider>
        </Grid>
      </Grid>

      <br />

      <Paper>
        <TableContainer>
          <Table stickyHeader aria-label="sticky table">
            <TableHead>
              <TableRow>
                {columns.map((column) => (
                  <TableCell
                    key={column.id}
                    align="left"
                    style={{ minWidth: column.minWidth, fontWeight: 'bold' }}
                  >
                    {column.label}
                  </TableCell>
                ))}
              </TableRow>
            </TableHead>
            <TableBody>
              {data?.length > 0 ? (
                data.map((row) => (
                  <TableRow key={row.DormName}>
                    {columns.map((column) => (
                      <TableCell key={column.id} align="left">
                        {column.id.includes('Ratio')
                          ? (typeof row[column.id] === 'number' && !isNaN(row[column.id])
                            ? `${parseFloat(row[column.id]).toFixed(1)}%`
                            : '')
                          : row[column.id]}
                      </TableCell>
                    ))}
                  </TableRow>
                ))
              ) : (
                <TableRow>
                  <TableCell colSpan={columns.length} align="center">
                    No data available
                  </TableCell>
                </TableRow>
              )}
            </TableBody>
          </Table>
        </TableContainer>
        <TablePagination
          rowsPerPageOptions={[10, 25, 100]}
          component="div"
          count={pageTotal}
          rowsPerPage={rowsPerPage}
          page={page}
          onPageChange={handleChangePage}
          onRowsPerPageChange={handleChangeRowsPerPage}
        />
      </Paper>
    </>
  );
}

export default Occupancyandvacancy;

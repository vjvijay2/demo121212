  /*?件箱*/
  PROCEDURE GET_SEND_LIST_SP(V_START_TIME  IN VARCHAR2,
                             V_END_TIME    IN VARCHAR2,
                             V_EMPNO       IN VARCHAR2,
                             V_EMPNAME     IN VARCHAR2,
                             V_FORM_CODE   IN VARCHAR2,
                             V_FORM_STATUS IN VARCHAR2,
                             V_USER        IN VARCHAR2,
                             P_CURSOR      OUT CURTYPE) AS
  BEGIN
    OPEN P_CURSOR FOR

      select DECODE(A.FORM_STATUS,
                    '0',
                    'Initiated',
                    '1',
                    'Signing',
                    '2',
                    'Finished',
                    '3',
                    'Rejected',
                    '4',
                    'Cancelled',
                    '') AS FORM_STATUS, --表???
             A.FORM_CODE,
             'LinkWindow$$$' || A.FORM_CODE LINK_FORM_CODE,
             A.Employee_Id as EMP_ID,
             A.Employee_Name AS EMP_NAME,
             A.DEPARTMENT,
             A.SHIFTING_DORM,
             A.REASON,
             A.Moved_By_Id AS ADD_BY,
             TO_CHAR(A.DORM_SHIFTED_DATE, 'DD-MM-YYYY HH24:MI:SS') AS DORM_SHIFTED_DATE
        FROM DMS.C_DORM_SHIFTING_T A
       WHERE 1 = 1
         AND (A.EMPLOYEE_ID = V_EMPNO OR V_EMPNO IS NULL) --可以查看自己的?据 也可以查看??人申?的?据
         AND (A.EMPLOYEE_NAME = V_EMPNAME OR V_EMPNAME IS NULL)
         AND (A.DORM_SHIFTED_DATE >=
             TO_DATE(V_START_TIME, 'DD-MM-YYYY HH24:MI:SS') OR
             V_START_TIME IS NULL)
         AND (A.DORM_SHIFTED_DATE <=
             TO_DATE(V_END_TIME, 'DD-MM-YYYY HH24:MI:SS') OR
             V_END_TIME IS NULL)
         AND A.Moved_By_Id = V_USER --?前?建人是自己 如果?建人?有??前申?人是自己
         AND (A.FORM_CODE = V_FORM_CODE OR V_FORM_CODE IS NULL)
         AND (A.FORM_STATUS = V_FORM_STATUS OR V_FORM_STATUS IS NULL);
  END;

  PROCEDURE GET_SIGN_LIST_SP(V_START_TIME  IN VARCHAR2,

  BEGIN
    OPEN P_CURSOR FOR
      select DECODE(A.FORM_STATUS,

  PROCEDURE GET_RECEIVE_COUNT_SP(V_START_TIME    IN VARCHAR2,
                                 V_END_TIME      IN VARCHAR2,
                                 V_EMPNO         IN VARCHAR2,
                                 V_EMPNAME       IN VARCHAR2,
                                 V_FORM_CODE     IN VARCHAR2,
                                 V_FORM_CODE_CON IN VARCHAR2,
                                 /* V_FORM_STATUS IN VARCHAR2,*/
                                 V_USER   IN VARCHAR2,
                                 P_CURSOR OUT CURTYPE) AS
  BEGIN
    OPEN P_CURSOR FOR
      SELECT V_FORM_CODE AS FORM_CODE
        FROM DUAL --必須加此句
      UNION ALL
      SELECT T.FORM_CODE
        FROM TN_APP.C_Domain_CREATION_T           T,
             WF_STRUCTURE.WF_FORM_OPT_CONDITION_T B
       WHERE B.FORM_CODE = T.FORM_CODE
         AND B.APPROVE_EMP_NO = V_USER
         AND (V_START_TIME IS NULL OR
             T.Created_Date >=
             to_date(V_START_TIME, 'DD-MM-YYYY HH24:MI:SS'))
         AND (V_END_TIME IS NULL OR
             T.Created_Date < TO_DATE(V_END_TIME, 'DD-MM-YYYY HH24:MI:SS'))
         AND (V_EMPNO IS NULL OR T.Emp_Id = V_EMPNAME)
         AND (V_EMPNAME IS NULL OR T.EMP_NAME = V_EMPNAME)
         AND T.FORM_CODE <> V_FORM_CODE --ADD BY XQS
         AND T.DEL_FLAG = '0'
         AND (V_FORM_CODE_CON IS NULL OR T.FORM_CODE = V_FORM_CODE_CON); --ADD ZF 2024/1/18 收件箱LIST 跳轉至表單詳情;
  END;
  PROCEDURE GET_FORM_SEARCH_INFO(V_FORM_CODE IN VARCHAR2,
                                 P_CURSOR    OUT CURTYPE) AS
  BEGIN
    OPEN P_CURSOR FOR
      SELECT A.Employee_Id AS EMP_ID, --工?
             A.Employee_NAME AS EMP_NAME, --姓名
             tn_emp.get_all_emp_base_info(A.MOVED_BY_ID, 20) AS SITE, --厂?
             tn_emp.get_all_emp_base_info(A.MOVED_BY_ID, 5) AS SALARY_GRADE, --?位
             tn_emp.get_all_emp_base_info(A.MOVED_BY_ID, 6) AS JOB_TITLE, --管理?
             A.DEPARTMENT, --部?
             tn_emp.get_all_emp_base_info(A.MOVED_BY_ID, 23) AS EXIT_NUMBER, --座机
             tn_emp.get_all_emp_base_info(A.MOVED_BY_ID, 11) AS PHONE, --手机
             tn_emp.get_all_emp_base_info(A.MOVED_BY_ID, 9) AS EMAIL, --?箱
             A.CURRENT_DOMR_ID,
             A.CURRENT_DORM,
             A.CURRENT_FLOOR,
             A.CURRENT_ROOM,
             A.CURRENT_BED,
             A.MOVED_BY_ID,
             A.SHIFTING_DORM,
             A.REASON
        FROM DMS.C_DORM_SHIFTING_T A
       WHERE A.FORM_CODE = V_FORM_CODE;
  END;
  PROCEDURE GET_FORM_DETAIL_INFO(V_FORM_CODE IN VARCHAR2,
                                 P_CURSOR    OUT CURTYPE) AS
  BEGIN
    OPEN P_CURSOR FOR
      SELECT A.APPLICATION_TYPE, A.REASON
        FROM TN_APP.C_Domain_CREATION_T A
       WHERE A.FORM_CODE = V_FORM_CODE
         AND A.DEL_FLAG = '0';
  END;

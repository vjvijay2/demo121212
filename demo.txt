import React, { useState, useEffect } from 'react';
import { Tabs } from 'antd';
import "./styles.css";
const { TabPane } = Tabs;

const ProgressSteps = ({  tableflow, engflowshow }) => {
  const [activeTab, setActiveTab] = useState('1');

  const callback = (key) => {
   // console.log('Tab changed. Key:', key);
    setActiveTab(key);
  };

  useEffect(() => {
    // Create signs when signFlow or signDetail updates
    createSign('#workFlow', tableflow);
    createSign('#workDetail', engflowshow);
  }, [tableflow, engflowshow]);

  const createSign = (idStr, content) => {
  //  console.log('Creating sign for', idStr);
    
    return (
      <div dangerouslySetInnerHTML={{ __html: content }} id={idStr} />
    );
  };

  const renderTabContent = (key) => {
    if (key === '1') {
    //  console.log('Rendering Tab 1 content');
      return createSign('workFlow', tableflow);
    } else if (key === '2') {
    //  console.log('Rendering Tab 2 content');
      return createSign('workDetail', engflowshow);
    }
    return null;
  };

  return (
    <div className="Signing">
      <Tabs onChange={callback} activeKey={activeTab}>
        <TabPane key="1" tab="WorkFlow">
          {renderTabContent('1')}
        </TabPane>
        <TabPane key="2" tab="WorkDetail">
          {renderTabContent('2')}
        </TabPane>
      </Tabs>
    </div>
  );
}

export default ProgressSteps;

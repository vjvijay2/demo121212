// api for fetchpmsemployeedata

export const fetchemployeefrompms = async (boardingpointcode) =>{
   
    const pageSize = 1000;
    const headers= {
        apiID:'ffff-1688525313868-1020819373-2274',
        userKey:userKey
    }

    try {
        const activeEmployees = [];
        const resignEmployees = [];
        let currentPage = 1;
    
         while (true) {
          const params = {
            Func: 'DetailsView-GetPersonalContactInfo',
            BoardingPointId: boardingpointcode,
            StartPage: currentPage,
            PageSize: pageSize,
          };
    //console.log(params)
          const encryptedParams = dealPostParamsEncrypt(params);
    
          const response = await axios.post(apiURL, encryptedParams, {
            headers,
           timeout: 10000,
          });
    
          const decryptedResponse = decrypt(response.data.r, getdk(iwxKey));
          const decryptedResponseObj = JSON.parse(decryptedResponse);
    
          const pageActiveEmployees = decryptedResponseObj.List.filter(
            (employee) => employee.EmpStatus === 'Active'
          );
          const pageResignEmployees = decryptedResponseObj.List.filter(
            (employee) =>
             employee.EmpStatus === 'Terminated' || employee.EmpStatus === 'Resigned' ||   employee.EmpStatus === 'Exit' 
        
          );
    
          activeEmployees.push(...pageActiveEmployees);
          resignEmployees.push(...pageResignEmployees);
    
          if (pageActiveEmployees.length + pageResignEmployees.length < pageSize) {
            break;
          }
    
           currentPage++;
        }
  // console.log(activeEmployees)
  // console.log(resignEmployees)
  
        return { activeEmployees, resignEmployees };
      } catch (error) {
        throw error;
      }
}

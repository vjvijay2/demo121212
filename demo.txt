<template>
  <div class="App">
    <van-tabs v-model="activeTab" type="card" class="custom-tabs" sticky @change="handleTabchange">
      <van-tab title="Shift-A" name="A"></van-tab>
      <van-tab title="Shift-B" name="B"></van-tab>
      <van-tab title="Shift-C" name="C"></van-tab>
      <van-tab title="Shift-G" name="G"></van-tab>
      <van-tab title="Shift-GN" name="GN"></van-tab>
      <van-tab title="Shift-GD" name="GD"></van-tab>
      <van-tab title="Shift-S" name="S"></van-tab>
    </van-tabs>

    <routed v-if="showChild" :details_data="details_data" />

    <van-search v-model="value" style="padding: 0px;" placeholder="Search Bus" shape="round" background="none" />

    <div v-if="loading">
      <van-loading class="loading-container" type="spinner" size="30px" color="#1989fa" />
    </div>

    <div v-else>
      <div v-if="list_item.length > 0">
        <div class="title">Bus Details</div>



        <div class="workForm">
          <van-collapse v-model="activeNames" class="custom-collapse">
            <van-collapse-item v-for="(routes, parentName) in groupedData" :key="parentName" :title="parentName"
              class="custom-collapse-item">
              <div v-for="route in routes" :key="route.id" class="child-route">

                <!-- STEP 1: Click to show basic info (Driver + Vehicle) -->
                <div v-if="!route.showBasic && !route.showFull" class="route-simple" @click="showBasicInfo(route)">
                  <div class="route-header-simple">
                    <span class="route-name">{{ route.ROUTE_NAME }}</span>
                    <span class="bus-number">{{ route.Busno }}</span>
                  </div>
                  <van-icon name="arrow-down" class="expand-icon" />
                </div>

                <!-- STEP 2: Basic info shown (Driver + Vehicle) - Click for full details -->
                <div v-else-if="route.showBasic && !route.showFull" class="route-basic-info">
                  <div class="basic-header">
                    <span class="route-name">{{ route.ROUTE_NAME }}</span>
                    <span class="bus-number">{{ route.Busno }}</span>
                  </div>

                  <div class="basic-details">
                    <div class="basic-item">
                      <van-icon name="user-o" />
                      <span>Driver: {{ route.drivername || 'Not Assigned' }}</span>
                    </div>
                    <div class="basic-item">
                      <van-icon name="logistics" />
                      <span>Vehicle: {{ route.VEHICLE_NUMBER || route.Busno }}</span>
                    </div>
                  </div>

                  <div class="tap-hint" @click="handleClick(route)">
                    <van-icon name="arrow-down" />
                    <span>Tap for full details</span>
                  </div>

                </div>


              </div>
            </van-collapse-item>
          </van-collapse>
        </div>











      </div>
      <div v-else style="padding-top: 100px;">
        <a-empty />
      </div>
    </div>

<script>
import { Cell, Popup, Picker, CellGroup, Field, Button, Icon, Loading } from 'vant';
import axios from "axios";
import CryptoJS from "crypto-js";
import routed from "./components/routes.vue";
import { ref } from "vue";

export default {
  components: {
    "van-cell": Cell,
    "van-popup": Popup,
    "van-picker": Picker,
    "van-cell-group": CellGroup,
    "van-field": Field,
    'van-icon': Icon,
    "van-loading": Loading,
    routed,
  },

  data() {
    return {
      activeTab: "A",
      loading: true,
      activeNames: [],
      FORM_CODE: "",
      formTypeId: '9889',
      list_item: [],
      Data: [], // Added: raw data storage
      details_data: "test",
      showChild: false,
      employee: {
        empId: "",
        name: "",
        department: "",
        jobtitle: "",
        OrgId: '',
        site: '',
      },
      error: "",
      iwxKey: "facd81g656fcd8p7a7lkh6dfa81gf4ik",
      strID: "",
      value: '',
      showPicker: false,
      showPopup: false,
      data: [
        { label: 'Employee ID', value: "", placeholder: 'Employee ID' },
        { label: 'Employee Name', value: '', placeholder: 'Enter Employee Name' },
        { label: 'Department', value: '', placeholder: 'Department' },
        { label: 'Employee Type', value: '', placeholder: 'Employee Type' },
      ],
      formData: [
        { label: 'Application Type', value: "", placeholder: 'Application type' },
        { label: 'Reason', value: '', placeholder: 'Reason' },
      ],
    };
  },

  computed: {
    // Moved from data() - now properly reactive
    groupedData() {
      const groupedRoutes = {};
      this.list_item.forEach(item => {
        const parent = item.ROUTE_NAME.split('-')[0].trim();
        if (!groupedRoutes[parent]) {
          groupedRoutes[parent] = [];
        }
        groupedRoutes[parent].push(item);
      });
      return groupedRoutes;
    }
  },


  watch: {
    value(newValue) {
      // Debounced search
      clearTimeout(this.searchTimer);
      this.searchTimer = setTimeout(() => {
        this.getformInfo();
      }, 300);
    },
  },

  methods: {


    showBasicInfo(route) {
      // Close other routes first
      this.list_item.forEach(r => {
        r.showBasic = false;
        r.showFull = false;
      });

      route.showBasic = true;
      route.showFull = false;
    },

    // Step 2: Show full details
    showFullDetails(route) {
      route.showFull = true;
    },

    // Collapse back to simple view
    collapseRoute(route) {
      route.showBasic = false;
      route.showFull = false;
    },

    callDriver(phone) {
      if (phone) {
        window.location.href = `tel:${phone}`;
      }
    },

    shareRoute(route) {
      // Share functionality
      const text = `Route: ${route.ROUTE_NAME}\nBus: ${route.Busno}\nDriver: ${route.drivername}\nContact: ${route.contactno}`;
      if (navigator.share) {
        navigator.share({ title: 'Bus Route Info', text });
      }
    },



    handleClick(item) {
      this.details_data = item;
      this.$router.push({
        path: `/routes`,           // ← Routes page
        query: {
          shift: this.activeTab,           // Current shift (A/B/C/G/GN/GD/S)
          routeName: item.ROUTE_NAME,      // Route name
          busNumber: item.Busno,           // Bus number
          driverName: item.drivername,     // Driver name
          driverNumber: item.contactno,    // Driver contact
          page: true
        }
      }).catch(()=>{})
    },


    getEmpNoByAPP() {
      // this.empId = "HH0013772";
        this.empId = "HH0023768";

      // this.empId = '';
      this.getUserBaseInfo(this.empId);

      window.callH5 = (data) => {
        data = JSON.parse(data);
        this.empId = data.empNo;
        this.getUserBaseInfo(this.empId);
      };
   
      this.$callNative.callNative2({ share: "key" });
    },

    getUserBaseInfo(empId) {
      this.$http.iwxPosts2({
        body: {
          EmployeeCode: empId,
          Func: "FIHManage-GetEMPBaseInfo"
        },
        isEncrypt: true,
        encryptKey: "facd81g656fcd8p7a7lkh6dfa81gf4ik",
        config: {
          headers: {
            apiID: "ffff-1669620301290-16925411121-5665",
            userKey: "C44F773FD87E7E41174D6A58B162B049",
          },
        },
      }).then((res) => {
        if (res.List && res.List.length > 0) {
          const responseData = res.List[0];
          this.employee = {
            empId: empId,
            site: responseData.SiteId,
          };
          this.getformInfo();
        } else {
          console.error("No employee data found");
          this.getformInfo();
   
        }
      }).catch((error) => {
        console.error(error);
      });
    },

    handleTabchange(tab) {
      this.list_item = [];
      this.activeTab = tab; // Fixed: was using == instead of =
      this.loading = true;
      setTimeout(() => {
        this.getformInfo();
      }, 500);
    },

    getformInfo() {
      this.$http.iwxPosts2({
        body: {
          V_SITE: this.employee.site.toString(),
          V_SHIFT: this.activeTab,
          V_BOARDING_POINT: this.value,

        },
        isEncrypt: false,
        config: {
          headers: {
            apiID: "ffff-1701083459163-10208193125-1065",
            userKey: "10A33A67A312E649D907BBC927995B8A"
          },
        },
      }).then((res) => {
        const dataArray = res.data;

        if (dataArray && dataArray.length > 0) {
          this.loading = false;
          this.Data = dataArray; // Store raw data
          this.list_item = dataArray.map((item) => ({
            id: item.id || item.ROUTE_NAME, // Added unique key
            ROUTE_NAME: item.ROUTE_NAME,
            Busno: item.VEHICLE_NUMBER,
            VEHICLE_NUMBER: item.VEHICLE_NUMBER,
            drivername: item.DRIVER_NAME,
            contactno: item.CONTACT_NUMBER,
            BOARDING_POINT: item.BOARDING_POINT,
            ROUTE_DETAILS: item.ROUTE_DETAILS,

            // 🔑 Two-step reveal flags
            showBasic: false,
            showFull: false,
          }));

        } else {
          console.error("No data found");
          this.loading = false;
          this.list_item = [];
          this.Data = [];
        }
      }).catch((error) => {
        console.error(error);
        this.loading = false;
      });
    },

    filteredData() {
      // Kept for compatibility, but logic moved to computed property
      this.list_item = this.Data.map((item) => ({
        ROUTE_NAME: item.ROUTE_NAME,
        Busno: item.VEHICLE_NUMBER,
        drivername: item.DRIVER_NAME,
        contactno: item.CONTACT_NUMBER,
      }));
    },
  },

  mounted() {
    setTimeout(() => {
      this.getEmpNoByAPP();
    }, 500);
  }
};
</script>

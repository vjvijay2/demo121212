    this.loadData();
  },

  watch: {
    '$route'(to) {
      this.loadData();
    }
  },

methods: {

  loadData() {
    const {
      shift,
      routeName,
      busNumber,
      driverName,
      driverNumber

this.getrouteinfo();

      headers: {
        apiID: "ffff-1701144856798-10208193125-1092",

      }
    };

    try {
      const res = await axios.post(
       
        "https://microservice.abgind.foxconn.com/india/gateway",
        params,
        config
      );
       console.log("response",res.data);

      const dataArray = res.data;
